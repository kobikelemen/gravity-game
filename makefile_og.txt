

CC = g++

INCLUDES = -I/usr/local/Cellar/sfml/2.5.1_1/include/

CFLAGS = -g $(INCLUDES)

LDFLAGS = -L/usr/local/Cellar/sfml/2.5.1_1/lib/


main: main.o game.o
	$(CC) main.o game.o $(LDFLAGS) -lsfml-graphics -lsfml-window -lsfml-system -o main

main.o: main.cpp
	$(CC) $(CFLAGS) -c main.cpp

game.o: game.cpp 
	$(CC) $(CFLAGS) -c game.cpp
